\makeatletter
\def\replicate#1#2{%
\@tempcnta=#1
\loop
	  #2
  \advance \@tempcnta -1
\ifnum \@tempcnta>0
\repeat
}
\makeatother


\begin{document}

\newcount\i
\begingroup
\replicate{5}{%
\advance\i by 1
\the\i%
}
\endgroup
